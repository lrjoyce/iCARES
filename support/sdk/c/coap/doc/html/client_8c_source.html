201 void
202 message_handler( coap_context_t *ctx, coap_queue_t *node, void *data) {
203  coap_pdu_t *pdu = NULL;
204  coap_opt_t *block, *ct, *sub;
205  unsigned int blocknr;
206  unsigned char buf[4];
207  coap_list_t *option;
208  unsigned int len;
209  unsigned char *databuf;
210 
211 #ifndef NDEBUG
212  printf("** process pdu: ");
213  coap_show_pdu( node->pdu );
214 #endif
215 
216  if ( node->pdu->hdr->version != COAP_DEFAULT_VERSION ) {
217  debug("dropped packet with unknown version %u\n", node->pdu->hdr->version);
218  return;
219  }
220 
221  if ( node->pdu->hdr->code < COAP_RESPONSE_100 && node->pdu->hdr->type == COAP_MESSAGE_CON ) {
222  /* send 500 response */
223  pdu = new_response( ctx, node, COAP_RESPONSE_500 );
224  goto finish;
225  }
226 
227  switch (node->pdu->hdr->code) {
228  case COAP_RESPONSE_200:
229  /* got some data, check if block option is set */
230  block = coap_check_option( node->pdu, COAP_OPTION_BLOCK );
231  if ( !block ) {
232  /* There is no block option set, just read the data and we are done. */
233  if ( coap_get_data( node->pdu, &len, &databuf ) ) {
234  /*path = coap_check_option( node->pdu, COAP_OPTION_URI_PATH );*/
235  append_to_file( "coap.out", databuf, len );
236  }
237  } else {
238  blocknr = coap_decode_var_bytes( COAP_OPT_VALUE(*block), COAP_OPT_LENGTH(*block) );
239 
240  /* TODO: check if we are looking at the correct block number */
241  if ( coap_get_data( node->pdu, &len, &databuf ) ) {
242  /*path = coap_check_option( node->pdu, COAP_OPTION_URI_PATH );*/
243  append_to_file( "coap.out", databuf, len );
244  }
245 
246  if ( (blocknr & 0x08) ) {
247  /* more bit is set */
248  printf("found the M bit, block size is %u, block nr. %u\n",
249  blocknr & 0x07,
250  (blocknr & 0xf0) << blocknr & 0x07);
251 
252  /* need to acknowledge if message was asyncronous */
253  if ( node->pdu->hdr->type == COAP_MESSAGE_CON ) {
254  pdu = new_ack( ctx, node );
255 
256  if ( pdu && coap_send( ctx, &node->remote, pdu ) == COAP_INVALID_TID ) {
257  debug("message_handler: error sending reponse");
258  coap_delete_pdu(pdu);
259  return;
260  }
261  }
262 
263  /* create pdu with request for next block */
264  pdu = coap_new_request( method, NULL ); /* first, create bare PDU w/o any option */
265  if ( pdu ) {
266  pdu->hdr->id = node->pdu->hdr->id; /* copy transaction id from response */
267 
268  /* get content type from response */
269  ct = coap_check_option( node->pdu, COAP_OPTION_CONTENT_TYPE );
270  if ( ct ) {
271  coap_add_option( pdu, COAP_OPTION_CONTENT_TYPE,
272  COAP_OPT_LENGTH(*ct),COAP_OPT_VALUE(*ct) );
273  }
274 
275  /* add URI components from optlist */
276  for (option = optlist; option; option = option->next ) {
277  switch (COAP_OPTION_KEY(*(coap_option *)option->data)) {
278  case COAP_OPTION_URI_AUTHORITY :
279  case COAP_OPTION_URI_PATH :
280  case COAP_OPTION_URI_QUERY :
281  coap_add_option ( pdu, COAP_OPTION_KEY(*(coap_option *)option->data),
282  COAP_OPTION_LENGTH(*(coap_option *)option->data),
283  COAP_OPTION_DATA(*(coap_option *)option->data) );
284  break;
285  default:
286  ; /* skip other options */
287  }
288  }
289 
290  /* finally add updated block option from response */
291  coap_add_option ( pdu, COAP_OPTION_BLOCK,
292  coap_encode_var_bytes(buf, blocknr + ( 1 << 4) ), buf);
293 
294  if ( coap_send_confirmed( ctx, &node->remote, pdu ) == COAP_INVALID_TID ) {
295  debug("message_handler: error sending reponse");
296  coap_delete_pdu(pdu);
297  }
298  return;
299  }
300  }
301  }

342  printf("** Humidity: NaN\n\n");
343  if (val.volt != (0xFFFF | 0xFFFE))
344  printf("** Voltage: %4.2f V\n\n", ((float) val.volt)/100);
345  else
346  printf("** Voltage: NaN\n\n");

348  } else if (strcmp((const char *)uri.path.s, "time") == 0) {

350  printf("** .well-known/core:\n");
351  } else {
352  printf("** unknown URI\n");
353  }
354 #endif
355 
356  break;
357  case COAP_RESPONSE_X_240:
358  printf("\n** Token required\n\n");
359  break;
360  case 0:
361  printf("** PreAck received\n");
362  break;
363  default:
364  ;
365  }
366 
367  /* acknowledge if requested */
368  if ( !pdu && node->pdu->hdr->type == COAP_MESSAGE_CON ) {
369  pdu = new_ack( ctx, node );
370  }
371 
372  finish:
373  if ( pdu && coap_send( ctx, &node->remote, pdu ) == COAP_INVALID_TID ) {
374  debug("message_handler: error sending reponse");
375  coap_delete_pdu(pdu);
376  }
377 
378  /* our job is done, we can exit at any time */
379  sub = coap_check_option( node->pdu, COAP_OPTION_SUBSCRIPTION );
380  if ( sub ) {
381  debug("message_handler: Subscription-Lifetime is %d\n",
382  COAP_PSEUDOFP_DECODE_8_4(*COAP_OPT_VALUE(*sub)));
383  }
384  ready = !sub || COAP_PSEUDOFP_DECODE_8_4(*COAP_OPT_VALUE(*sub)) == 0;
385 }

412 int
413 join( coap_context_t *ctx, char *group_name ){
414  struct ipv6_mreq mreq;
415  struct addrinfo *reslocal = NULL, *resmulti = NULL, hints, *ainfo;
416  int result = -1;
417 
418  /* we have to resolve the link-local interface to get the interface id */
419  memset(&hints, 0, sizeof(hints));
420  hints.ai_family = AF_INET6;
421  hints.ai_socktype = SOCK_DGRAM;
422 
423  result = getaddrinfo("::", NULL, &hints, &reslocal);
424  if ( result < 0 ) {
425  perror("join: cannot resolve link-local interface");
426  goto finish;
427  }
428 
429  /* get the first suitable interface identifier */
430  for (ainfo = reslocal; ainfo != NULL; ainfo = ainfo->ai_next) {
431  if ( ainfo->ai_family == AF_INET6 ) {
432  mreq.ipv6mr_interface =
433  ((struct sockaddr_in6 *)ainfo->ai_addr)->sin6_scope_id;
434  break;
435  }
436  }
437 
438  memset(&hints, 0, sizeof(hints));
439  hints.ai_family = AF_INET6;
440  hints.ai_socktype = SOCK_DGRAM;
441 
442  /* resolve the multicast group address */
443  result = getaddrinfo(group_name, NULL, &hints, &resmulti);
444 
445  if ( result < 0 ) {
446  perror("join: cannot resolve multicast address");
447  goto finish;
448  }
449 
450  for (ainfo = resmulti; ainfo != NULL; ainfo = ainfo->ai_next) {
451  if ( ainfo->ai_family == AF_INET6 ) {
452  mreq.ipv6mr_multiaddr =
453  ((struct sockaddr_in6 *)ainfo->ai_addr)->sin6_addr;
454  break;
455  }
456  }
457 
458  result = setsockopt( ctx->sockfd, IPPROTO_IPV6, IPV6_JOIN_GROUP,
459  (char *)&mreq, sizeof(mreq) );
460  if ( result < 0 )
461  perror("join: setsockopt");
462 
463  finish:
464  freeaddrinfo(resmulti);
465  freeaddrinfo(reslocal);
466 
467  return result;
468 }
