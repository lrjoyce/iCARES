33 int
34 coap_split_uri(unsigned char *str_var, coap_uri_t *uri) {
35  unsigned char *p;
36 
37  if ( !str_var || !uri )
38  return -1;
39 
40  memset( uri, 0, sizeof(coap_uri_t) );
41 
42  /* find scheme */
43  p = str_var;
44  while ( isalnum(*p) )
45  ++p;
46 
47  if ( *p != ':' ) { /* no scheme, reset p */
48  p = str_var;
49  } else { /* scheme found, check if it is "coap" */
50  if (memcmp(str_var, COAP_DEFAULT_SCHEME,
51  MIN(p - str_var, sizeof(COAP_DEFAULT_SCHEME) - 1)) != 0) {
52 #ifndef NDEBUG
53  debug("unknown URI scheme '%s'\n", str_var);
54 #endif
55  return -1;
56  }
57  *p++ = '\0';
58 
59  /* look for network authority */
60  if ( strncmp( (char *)p, "//", 2 ) == 0 ) { /* have network authority */
61  p += 2;
62  uri->na.s = p;
63 
64  /* skip NA and port so that p and str_var finally point to path */
65  while ( *p && *p != '/' && *p != '?')
66  ++p;
67 
68  uri->na.length = p - uri->na.s;
69 
70  str_var = p;
71 #if 0
72  /* split server address and port */
73  if ( *uri->na == '[' ) { /* IPv6 address reference */
74  p = ++uri->na;
75 
76  while ( *p && *p != ']' )
77  ++p;
78  *p++ = '\0';
79  } else { /* IPv4 address or hostname */
80  p = uri->na;
81  while ( *p && *p != ':' )
82  ++p;
83  }
84 
85  if ( *p == ':' ) { /* handle port */
86  *p++ = '\0';
87  uri->port = p;
88  }
89 #endif
90  } else
91  str_var = p;
92 
93  /* str_var now points to the path or query if path is empty*/
94  }
95 
96  /* split path and query */
97 
98  if ( *str_var == '\0' )
99  return 0;
100 
101  if (*str_var != '?') {
102  if (*str_var == '/') /* skip leading '/' */
103  *str_var++ = '\0';
104  uri->path.s = str_var;
105  }
106 
107  while (*str_var && *str_var != '?')
108  str_var++;
109 
110  if (*str_var == '?') {
111  *str_var++ = '\0';
112 
113  if (*str_var) {
114  uri->query.s = str_var;
115  uri->query.length = strlen((char *)uri->query.s);
116  }
117  }
118 
119  if (uri->path.s)
120  uri->path.length = strlen((char *)uri->path.s);
121 
122  return 0;
123 }

127 coap_uri_t *
128 coap_new_uri(const unsigned char *uri, unsigned int length) {
129  unsigned char *result = coap_malloc(length + 1 + sizeof(coap_uri_t));
130  if ( !result )
131  return NULL;
132 
133  memcpy(URI_DATA(result), uri, length);
134  URI_DATA(result)[length] = '\0'; /* make it zero-terminated */
135 
136  coap_split_uri( URI_DATA(result), (coap_uri_t *)result );
137  return (coap_uri_t *)result;
138 }

141 coap_clone_uri(const coap_uri_t *uri) {

143 
144  if ( !uri )
145  return NULL;
146 
147  result = (coap_uri_t *)coap_malloc( uri->query.length + uri->na.length +
148  uri->path.length + sizeof(coap_uri_t) + 1);
149 
150  if ( !result )
151  return NULL;
152 
153  memset( result, 0, sizeof(coap_uri_t) );
154 
155  if ( uri->na.length ) {
156  result->na.s = URI_DATA(result);
157  result->na.length = uri->na.length;
158 
159  memcpy(result->na.s, uri->na.s, uri->na.length);
160  }
161 
162  if ( uri->path.length ) {
163  result->path.s = URI_DATA(result) + uri->na.length;
164  result->path.length = uri->path.length;
165 
166  memcpy(result->path.s, uri->path.s, uri->path.length);
167  }
168 
169  if ( uri->query.length ) {
170  result->query.s = URI_DATA(result) + uri->na.length + uri->path.length;
171  result->query.length = uri->query.length;
172 
173  memcpy(result->query.s, uri->query.s, uri->query.length);
174  }
